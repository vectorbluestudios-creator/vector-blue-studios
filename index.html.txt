        let cart = {}; // Store as { appId: quantity }

        function renderApps(filter = 'all') {
            const grid = document.getElementById('app-grid');
            grid.innerHTML = '';
            const filteredApps = filter === 'all' ? apps : apps.filter(app => app.category === filter);

            filteredApps.forEach(app => {
                const card = document.createElement('div');
                card.className = 'app-card rounded-2xl p-6 flex flex-col h-full relative overflow-hidden group';
                const bgPattern = `<div class="absolute -right-10 -top-10 w-32 h-32 bg-[#0051ff] rounded-full blur-3xl opacity-10 group-hover:opacity-30 transition duration-500"></div>`;
                card.innerHTML = `
                    ${bgPattern}
                    <div class="flex justify-between items-start mb-4 z-10 relative">
                        <span class="text-xs font-bold tracking-wider text-[#00f0ff] uppercase">${app.tag}</span>
                        <span class="text-xs text-gray-500 uppercase border border-gray-700 px-2 py-1 rounded">${app.category}</span>
                    </div>
                    <h3 class="text-2xl font-bold mb-3 z-10 relative">${app.name}</h3>
                    <p class="text-gray-400 text-sm mb-8 flex-grow z-10 relative leading-relaxed">${app.desc}</p>
                    <div class="flex justify-between items-center mt-auto z-10 relative pt-4 border-t border-gray-800">
                        <span class="text-lg font-bold text-white">${app.priceStr}</span>
                        <button onclick="addToCart(${app.id})" class="btn-glow text-white text-sm font-bold py-2 px-5 rounded-full">
                            Add to Cart
                        </button>
                    </div>
                `;
                grid.appendChild(card);
            });
        }

        function filterApps(category) {
            document.querySelectorAll('.category-btn').forEach(btn => {
                btn.classList.remove('active');
                if(btn.innerText.toLowerCase() === category.toLowerCase()) btn.classList.add('active');
            });
            renderApps(category);
        }

        /* --- CART LOGIC --- */
        function addToCart(appId) {
            if (cart[appId]) {
                cart[appId]++;
            } else {
                cart[appId] = 1;
            }
            updateCartUI();
            
            // Notification pop animation
            const counter = document.getElementById('cart-count');
            counter.classList.add('scale-150');
            setTimeout(() => counter.classList.remove('scale-150'), 200);
        }

        function changeQuantity(appId, delta) {
            if (cart[appId]) {
                cart[appId] += delta;
                if (cart[appId] <= 0) delete cart[appId];
                updateCartUI();
            }
        }

        function clearCart() {
            cart = {};
            updateCartUI();
        }

        function updateCartUI() {
            const cartItemsContainer = document.getElementById('cart-items');
            cartItemsContainer.innerHTML = '';
            
            let totalItems = 0;
            let totalPrice = 0;

            if (Object.keys(cart).length === 0) {
                cartItemsContainer.innerHTML = '<p class="text-gray-500 text-center mt-10">Your cart is empty.</p>';
            } else {
                Object.keys(cart).forEach(id => {
                    const app = apps.find(a => a.id == id);
                    const qty = cart[id];
                    totalItems += qty;
                    totalPrice += (app.price * qty);

                    cartItemsContainer.innerHTML += `
                        <div class="flex justify-between items-center bg-[#111] p-4 rounded-xl border border-gray-800">
                            <div class="flex-1">
                                <h4 class="font-bold text-sm text-white">${app.name}</h4>
                                <span class="text-xs text-[#00f0ff]">$${app.price.toFixed(2)}</span>
                            </div>
                            <div class="flex items-center gap-3 bg-[#050505] rounded-lg px-2 py-1 border border-gray-700">
                                <button onclick="changeQuantity(${app.id}, -1)" class="text-gray-400 hover:text-white px-1">-</button>
                                <span class="text-sm font-bold w-4 text-center">${qty}</span>
                                <button onclick="changeQuantity(${app.id}, 1)" class="text-gray-400 hover:text-white px-1">+</button>
                            </div>
                        </div>
                    `;
                });
            }

            document.getElementById('cart-count').innerText = totalItems;
            document.getElementById('cart-total').innerText = `$${totalPrice.toFixed(2)}`;
        }

        function toggleCart(event) {
            // Prevent overlay click from triggering if clicking inside the panel
            if (event && event.target.id !== 'cart-overlay' && event.currentTarget.id === 'cart-overlay') return;
            
            const overlay = document.getElementById('cart-overlay');
            const panel = document.getElementById('cart-panel');
            
            if (overlay.classList.contains('modal-enter-active')) {
                panel.classList.remove('slide-panel-active');
                setTimeout(() => overlay.classList.remove('modal-enter-active'), 300);
            } else {
                overlay.classList.add('modal-enter-active');
                setTimeout(() => panel.classList.add('slide-panel-active'), 10);
            }
        }

        /* --- CHECKOUT LOGIC --- */
        function processCheckout() {
            if (Object.keys(cart).length === 0) {
                alert("Your cart is empty!");
                return;
            }

            // Calculate totals for receipt
            const receiptItems = document.getElementById('receipt-items');
            let totalPrice = 0;
            receiptItems.innerHTML = '';

            Object.keys(cart).forEach(id => {
                const app = apps.find(a => a.id == id);
                const qty = cart[id];
                const lineTotal = app.price * qty;
                totalPrice += lineTotal;
                
                receiptItems.innerHTML += `
                    <div class="flex justify-between text-sm py-2 border-b border-gray-800 last:border-0">
                        <span class="text-gray-300">${qty}x ${app.name}</span>
                        <span class="text-white">$${lineTotal.toFixed(2)}</span>
                    </div>
                `;
            });

            document.getElementById('receipt-total').innerText = `$${totalPrice.toFixed(2)}`;

            // Hide Cart and Show Receipt
            toggleCart(); // Close cart
            document.getElementById('receipt-modal').classList.add('modal-enter-active');
        }

        function closeReceipt() {
            document.getElementById('receipt-modal').classList.remove('modal-enter-active');
            clearCart(); // Empty the cart after successful purchase
        }

        // Initialize
        renderApps();
        updateCartUI(); // Initial blank state for cart
